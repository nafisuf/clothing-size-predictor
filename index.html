// Contoh fetch ke API
async function cekUkuran(tinggiCm, beratKg, usiaTahun) {
  const res = await fetch('https://api-anda.onrender.com/predict', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ height: tinggiCm, weight: beratKg, age: usiaTahun })
  });
  const data = await res.json();
  console.log(`Ukuran: ${data.predicted_size}, Confidence: ${data.confidence*100}%`);
  return data;
}
